    <script>
        const app = {
            init() {
                lucide.createIcons();
                this.initCharts();
            },
            switchTab(tabId) {
                document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
                document.getElementById(`tab-${tabId}`).classList.add('active');
                document.querySelectorAll('.nav-btn').forEach(btn => {
                    btn.classList.remove('text-emerald-600', 'border-emerald-500');
                    btn.classList.add('text-slate-500');
                    if(btn.classList.contains('nav-' + tabId)) {
                        btn.classList.add('text-emerald-600', 'border-emerald-500');
                    }
                });
                setTimeout(() => window.dispatchEvent(new Event('resize')), 100);
            },
            initCharts() {
                const ctxMath = document.getElementById('mathSimChart').getContext('2d');
                new Chart(ctxMath, {
                    type: 'line',
                    data: {
                        labels: ['10:00','10:05','10:10','10:15','10:20'],
                        datasets: [
                            { label: 'p90 (Estrés)', data: [85, 92, 88, 95, 99], borderColor: '#10b981', fill: false, tension: 0.4 },
                            { label: 'Realidad', data: [70, 75, 72, 80, 82], borderColor: '#94a3b8', borderDash: [5,5], fill: false }
                        ]
                    },
                    options: { responsive: true, maintainAspectRatio: false }
                });

                const ctxEco = document.getElementById('ecoRoiChart').getContext('2d');
                new Chart(ctxEco, {
                    type: 'bar',
                    data: {
                        labels: ['Q1', 'Q2', 'Q3', 'Q4'],
                        datasets: [{ label: 'Retorno Neto (%)', data: [12, 28, 45, 68], backgroundColor: '#10b981' }]
                    },
                    options: { responsive: true, maintainAspectRatio: false }
                });
            }
        };
        window.app = app;
        window.onload = () => app.init();
    </script>
